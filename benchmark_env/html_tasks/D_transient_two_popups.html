#!/usr/bin/env python3
"""Main entry point for Project Alpha pipeline."""

import os
import sys
import json
from pathlib import Path
from utils import load_config, setup_logger

# Configuration
CONFIG_PATH = Path("config.json")
LOG_LEVEL = os.getenv("LOG_LEVEL", "INFO")

def initialize_pipeline(config: dict) -> bool:
    """Initialize the data pipeline with configuration."""
    logger = setup_logger(LOG_LEVEL)
    logger.info(f"Starting pipeline v{config['version']}")

    try:
        stages = config.get("stages", [])
        for stage in stages:
            logger.debug(f"Processing stage: {stage['name']}")
            run_stage(stage)
        return True
    except Exception as e:
        logger.error(f"Pipeline failed: {e}")
        return False

def main():
    config = load_config(CONFIG_PATH)
    success = initialize_pipeline(config)
    sys.exit(0 if success else 1)

if __name__ == "__main__":
    main()
